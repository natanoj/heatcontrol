\begin{abstract}
\setcounter{page}{1}
\noindent{This document describes the specification, planning,
  designing, building and testing of an electronic domestic heat
  control system. The system regulates the temperature in rooms in a
  house by measuring room and outdoor temperature and regulating
  temperature of water based convector heaters as well as water based
  floor heating.}

\end{abstract}
